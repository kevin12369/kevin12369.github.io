        if not root:
            return 0
        
        if root.val not in '+-*/':
            return int(root.val)
        
        left = self.evaluate(root.left)
        right = self.evaluate(root.right)
        
        if root.val == '+':
            return left + right
        elif root.val == '-':
            return left - right
        elif root.val == '*':
            return left * right
        elif root.val == '/':
            return left // right
